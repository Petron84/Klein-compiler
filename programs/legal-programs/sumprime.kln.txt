* This program defines a helper function IsPrime(n) to test for primality and a main function sumPrimes(limit) that adds all primes  
*  <=limit. Prints the result for limit =20.

function isPrime(n)
    if n < 2 then
        return false
    end
    i := 2
    while i * i <= n do
        if n mod i == 0 then
            return false
        end
        i := i + 1
    end
    return true
end

function sumPrimes(limit)
    total := 0
    num := 2
    while num <= limit do
        if isPrime(num) then
            total := total + num
        end
        num := num + 1
    end
    return total
end

print(sumPrimes(20))
